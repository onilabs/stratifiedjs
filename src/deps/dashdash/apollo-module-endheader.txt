var format = function(str /*, replacements ... */) {
    // upstream uses node.js' util/format function,
    // which isn't available in the browser.
    var str = arguments[0];
    var idx = 1;
    var args = arguments;
    return str.replace(/%(.)/g, function(text, fmt) {
        var obj = args[idx++];
        if (fmt == 'j') return JSON.stringify(obj);
        return String(obj);
    });
}
